#version 430
#extension GL_NV_shadow_samplers_cube : enable

layout(location=2) uniform vec4 lightPos;
layout(location=3) uniform samplerCube skybox;


in vec4 fcolor;
in vec4 fpos;
in vec4 fnorm;

flat in int fTextType;

out vec4 gli_FragColor;

void main() {	

	vec4 textureColor = vec4(1.0f,0.0f,0.0f,1.0f);

	if(fTextType==0){
		textureColor = texture(skybox,normalize(fpos.xyz));
	}else if(fTextType==1){
		//textureColor = texture2D(skybox, tempCoords);
	}else if(fTextType==2){
		textureColor = texture(skybox,normalize(fpos.xyz));
	}	


	gli_FragColor = textureColor;

	
}
